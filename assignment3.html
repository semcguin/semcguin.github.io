def extract_fresh(f, type):
    x = pd.read_excel(f)
    if x.iloc[2,0] == "Fresh1":
        fresh = x.iloc[2]
        fresh = fresh.drop(fresh.index[[2,5]]) 
        fresh['7'] = fresh.index[[0]]
        fresh['8'] = type
        return fresh
    else:
        return []
